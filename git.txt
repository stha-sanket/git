git init
- starts git in that folder(repo/repository)
- .git file is created

git config --global user.name "username"
- configure that the user name to all git system

git config --local user.name "username"
- configure that the user name to all git system

git config --global user.email "email"
- configure that the email to all git system

git config --global init.defaultBranch main
- makes so that everytime we create .git file the branch name will be "main"

git status
- shows us branch, shows us our commit, shows staged area, shows uncommit file

git add file name
- add file to commit

git branch
- shows us branch

git branch [branch-name]
- create a new branch at the current commit
